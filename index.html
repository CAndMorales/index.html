1D Random Walk¶

A particle move in a 1D space: In one dimension the particle move up or down randomly with equal probability. The walks occurs for a given number of steps.

There is a reflecting barrier - if the particle collides with the reflecting barrier, the next step is up. There is also an absorbing barrier - if the particle collides with the abasorbing barrier the walk ends (the particle disappears).

The input (independent) variables of the system: this is how you set up your system.

The output (dependent) variables of the system: this is what you observe according to the system setup.

def make_walk(absorbing_barrier, reflecting_barrier = 0, start = 0):
    
    current_index = 0
    end = False
    
    walk = [start]
    
    while end != True:
        
        # Si la partcula colisiona con la barrera reflectiva, el siguiente paso es 1
        if walk[current_index] == reflecting_barrier:
            walk.append(1)
        
        # Si la partcula colisiona con la barrera absorbente, el camino finaliza
        elif walk[current_index] == absorbing_barrier:
            walk.append(walk[current_index])
            end = True
        
        # Para cada paso se suma o resta 1 segn la funcin de probabilidad uniforme
        else:
            walk.append(walk[current_index] + random_value())
    
        current_index += 1
    
    return walk

def average_steps(absorbing_barrier = 4, num_trials = 1000):
    
    # Crea un vector vaco de tamao N
    len_walks = np.empty(num_trials, dtype=object)
    
    for i in range(num_trials):
    
        # Realiza una simulacin del camino aleatorio en una dimensin
        walk = make_walk(absorbing_barrier)
        
        # Asigna la cantidad de pasos que le tom llegar a la barrera absorbente
        len_walks[i] = len(walk)

    # Calcula el promedio
    value = (np.mean(len_walks)).astype(int)
    
    return value

def min_max_steps(absorbing_barrier = 4, num_trials = 1000):
    
    # Crea un vector vaco de tamao N
    len_walks = np.empty(num_trials, dtype=object)
    
    for i in range(num_trials):
    
        # Realiza una simulacin del camino aleatorio en una dimensin
        walk = make_walk(absorbing_barrier)
        
        # Asigna la cantidad de pasos que le tom llegar a la barrera absorbente
        len_walks[i] = len(walk)

    # Obtiene la mnima cantidad de pasos necesarios para llegar a la barrera absorbente
    min = np.amin(len_walks)
    
    # Obtiene la mxima cantidad de pasos necesarios para llegar a la barrera absorbente
    max = np.amax(len_walks)
    
    return [min, max]
